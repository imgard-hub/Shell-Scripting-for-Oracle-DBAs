If-Else Statements

Basic if / elif / else
# Simple if
if [ $PCT -gt 85 ]; then
    echo "WARNING"
fi

# If-else
if [ $PCT -gt 85 ]; then
    echo "WARNING"
else
    echo "OK"
fi

# Multiple conditions with elif
if [ $PCT -ge 95 ]; then
    echo "🔴 CRITICAL: ${PCT}% — act NOW"
elif [ $PCT -ge 85 ]; then
    echo "🟡 WARNING: ${PCT}% — plan to add space"
elif [ $PCT -ge 70 ]; then
    echo "🟢 MONITOR: ${PCT}% — watch growth"
else
    echo "✅ OK: ${PCT}%"
fi
DBA example: check database status
#!/bin/bash
STATUS=$(sqlplus -s / as sysdba <<EOF
SET HEADING OFF FEEDBACK OFF
SELECT status FROM v\$instance;
EXIT;
EOF
)

if [ "$STATUS" = "OPEN" ]; then
    echo "✅ Database is UP"
else
    echo "🚨 Database is DOWN: $STATUS"
    # send alert email here
fi
