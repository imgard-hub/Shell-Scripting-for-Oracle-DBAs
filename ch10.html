Case Statements

When you have many possible values, case is cleaner than chaining if-elif-elif. Great for script arguments and menu options.

DBA example: multi-action admin tool
#!/bin/bash
# Usage: ./db_admin.sh start|stop|status|backup

case "$1" in
    start)
        echo "Starting database..."
        sqlplus / as sysdba <<EOF
        STARTUP;
        EXIT;
EOF
        ;;
    stop)
        echo "Stopping database..."
        sqlplus / as sysdba <<EOF
        SHUTDOWN IMMEDIATE;
        EXIT;
EOF
        ;;
    status)
        sqlplus -s / as sysdba <<EOF
        SELECT instance_name, status FROM v\$instance;
        EXIT;
EOF
        ;;
    backup)
        rman target / <<EOF
        BACKUP DATABASE PLUS ARCHIVELOG;
        EXIT;
EOF
        ;;
    *)
        echo "Usage: $0 {start|stop|status|backup}"
        exit 1
        ;;
esac
