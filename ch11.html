Loops

For loop — iterate a list
# Check all databases
for DB in PRODDB TESTDB DEVDB; do
    echo "Checking: $DB"
    export ORACLE_SID=$DB
    sqlplus -s / as sysdba <<EOF
    SELECT status FROM v\$instance;
    EXIT;
EOF
done

# Loop through files (cleanup old logs)
for FILE in $(find /home/oracle/logs -name "*.log" -mtime +30); do
    echo "Removing: $FILE"
    rm "$FILE"
done
While loop — repeat until condition changes
# Wait for database to come up (with timeout)
COUNTER=0
MAX_WAIT=10

while [ $COUNTER -lt $MAX_WAIT ]; do
    STATUS=$(sqlplus -s / as sysdba <<EOF
SET HEADING OFF FEEDBACK OFF
SELECT status FROM v\$instance;
EXIT;
EOF
)
    if [ "$STATUS" = "OPEN" ]; then
        echo "Database is OPEN!"
        break
    fi
    echo "Waiting... $COUNTER/$MAX_WAIT"
    sleep 30
    COUNTER=$((COUNTER + 1))
done
Nested loops — check multiple things across databases
for DB in PRODDB TESTDB DEVDB; do
    echo "=== $DB ==="
    export ORACLE_SID=$DB
    for CHECK in status space backup; do
        echo "  Running: $CHECK"
    done
done
Real-world use: Nested loops let you build one health-check script that monitors ALL databases across ALL checks — one script, one cron job, complete coverage.
