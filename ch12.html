Functions

Functions let you write code once and call it anywhere. They keep scripts clean, short, and maintainable.

Defining and using functions
#!/bin/bash

# Logging function
log() {
    echo "[$(date '+%H:%M:%S')] $1" | tee -a $LOG_FILE
}

# Database check function
check_db() {
    local SID=$1           # local keeps var inside function
    export ORACLE_SID=$SID
    STATUS=$(sqlplus -s / as sysdba <<EOF
SET HEADING OFF FEEDBACK OFF
SELECT status FROM v\$instance;
EXIT;
EOF
)
    if [ "$STATUS" = "OPEN" ]; then
        log "$SID: UP"
    else
        log "$SID: DOWN — $STATUS"
    fi
}

# Call the function for each database
LOG_FILE="/home/oracle/logs/check.log"
log "Starting health check"
check_db PRODDB
check_db TESTDB
check_db DEVDB
log "Complete"
