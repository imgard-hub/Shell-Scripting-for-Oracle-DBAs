Working with Files & Output

Redirection — controlling where output goes
# >   Write to file (overwrite)
echo "Report started" > report.txt

# >>  Append to file
echo "Line 2" >> report.txt

# 2>  Redirect errors
sqlplus / as sysdba @check.sql 2> errors.log

# &>  Redirect BOTH output and errors
sqlplus / as sysdba @check.sql &> full_output.log

# |   Pipe output into another command
cat alert.log | grep "ORA-" | tail -20
DBA example: build a daily report file
#!/bin/bash
REPORT="/home/oracle/reports/daily_$(date +%Y%m%d).txt"

echo "=== Daily DBA Report ==="       > $REPORT
echo "Date: $(date)"                 >> $REPORT
echo "Server: $(hostname)"            >> $REPORT
echo ""                               >> $REPORT
echo "--- Disk Space ---"             >> $REPORT
df -h                                >> $REPORT
echo ""                               >> $REPORT
echo "--- Alert Log Errors ---"       >> $REPORT
grep -i "ORA-" $ALERT_LOG | tail -10 >> $REPORT

# Email the report
mail -s "Daily Report - $(date +%Y-%m-%d)" $EMAIL < $REPORT
